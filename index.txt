    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const downloadBtn = document.getElementById('downloadBtn');
            const modal = document.getElementById('disclaimerModal');
            const acceptBtn = document.getElementById('acceptBtn');
            const declineBtn = document.getElementById('declineBtn');

            if (downloadBtn && modal && acceptBtn && declineBtn) {
                // Function to show the modal
                const showModal = (e) => {
                    e.preventDefault(); // Prevent the link from navigating
                    modal.classList.add('visible');
                };

                // Function to hide the modal
                const hideModal = () => {
                    modal.classList.remove('visible');
                };

                // Show modal on download button click
                downloadBtn.addEventListener('click', showModal);

                // Hide modal on decline
                declineBtn.addEventListener('click', hideModal);

                // Hide modal and proceed with download on accept
                acceptBtn.addEventListener('click', () => {
                    hideModal();
                    // Proceed to the download link
                    window.open(downloadBtn.href, '_blank');
                });

                // Optional: Hide modal if user clicks on the overlay
                modal.addEventListener('click', (e) => {
                    if (e.target === modal) {
                        hideModal();
                    }
                });
            }
        });
    </script>
